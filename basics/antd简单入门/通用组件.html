# 2.通用组件

# 2.1.按钮

# 按钮类型

type值：

primary
dashed
danger
link

首先需要引入Button按钮组件

import { Button } from 'antd';

{/* 默认样式 */}
<Button>按钮</Button>
{/* 蓝色按钮 */}
<Button type="primary">按钮</Button>
{/* 红色警告 */}
<Button type="danger">按钮</Button>
{/* 虚线 */}
<Button type="dashed">按钮</Button>
{/* 链接样式 */}
<Button type="link" >按钮</Button>

# 图标按钮

{/* 图标按钮 */}
<Button icon="search">按钮</Button>

你可以在官方文档上搜索 icon组件支持的图标名称，然后填入Button按钮的icon属性即可，如果你想修改图标的位置，不能使用属性的方式加载图标，需要把图标组件引入到Button中，如下：

<Button><Icon type="home"></Icon>按钮</Button>

# 按钮尺寸

有三个尺寸，大、中、小，默认为中，大为large 小为small

<Button size="large">按钮</Button>
<Button >按钮</Button>
<Button size="small">按钮</Button>

# 不可用状态

<Button disabled>按钮</Button>

# 加载中状态

<Button loading>按钮</Button>

# 下拉选项按钮

import React from 'react';
import './App.css';
import { Button,Menu, Icon, Dropdown } from 'antd'

function App() {
  const menu = (
    <Menu >
      <Menu.Item key="1">1st item</Menu.Item>
      <Menu.Item key="2">2nd item</Menu.Item>
      <Menu.Item key="3">3rd item</Menu.Item>
    </Menu>
  );
  return (
    <div className="App">
      <header className="App-header">
        
        <Dropdown overlay={menu}>
          <Button>
            Actions <Icon type="down" />
          </Button>
        </Dropdown>
      </header>
    </div>
  );
}

export default App;

# 按钮组

<Button.Group>
    <Button >L</Button>
    <Button >M</Button>
    <Button >R</Button>
</Button.Group>

# 幽灵按钮

幽灵按钮将按钮的内容反色，背景变为透明，常用在有色背景上。

<Button type="danger" ghost>
    danger
</Button>

# block按钮

<Button type="danger" block>
    Danger
</Button>

# onClick

import React from 'react';
import './App.css';
import { Button } from 'antd'

class App extends React.Component{
  handleClick = () => {
    alert(11)
  }
  render(){
    return (
      <div className="App">
        <header className="App-header">
          <Button type="danger" block onClick={this.handleClick}>
            danger
          </Button>
        </header>
      </div>
    )
  }
}


export default App;

# 总结：

多个属性的设置顺序：type -> shape -> size -> loading -> disabled

下面的关键词可以简单记忆一下：

disabled 按钮失效状态
ghost  幽灵按钮
href   点击跳转的地址，指定此属性 button 的行为和 a 链接一致
htmlType 设置 button 原生的 type 值
icon  设置按钮的图标类型
loading 设置按钮载入状态
shape   设置按钮形状，可选值为 circle、 round 或者不设
size    设置按钮大小，可选值为 small large 或者不设
target  相当于 a 链接的 target 属性，href 存在时生效
type    设置按钮类型，可选值为 primary dashed danger link(3.17 中增加) 或者不设
onClick  点击按钮时的回调
block   将按钮宽度调整为其父宽度的选项

# 总结

下面属性可以简单记忆

type	图标类型。遵循图标的命名规范
style	设置图标的样式，例如 fontSize 和 color
theme	图标主题风格。可选实心、描线、双色等主题风格，适用于官方图标
spin	是否有旋转动画
rotate	图标旋转角度（3.13.0 后新增，IE9 无效）
component	控制如何渲染图标，通常是一个渲染根标签为 <svg> 的 React 组件，会使 type 属性失效
twoToneColor	仅适用双色图标。设置双色图标的主要颜色

# 标题组件

import React from 'react';

import { Typography } from 'antd'



class App extends React.Component {

  render() {
    const { Title } = Typography;
    return (
      <div className="App">
        <header className="App-header">
          <Title>h1. Ant Design</Title>
          <Title level={2}>h2. Ant Design</Title>
          <Title level={3}>h3. Ant Design</Title>
          <Title level={4}>h4. Ant Design</Title>
         
        </header>
      </div>
    )
  }
}


export default App;
