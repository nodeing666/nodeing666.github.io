2.数据持久化处理

我们做数据持久化处理就是把数据存储在localstorage里面，主要步骤有三步：

第一步，写一个TodoModel.js类

export default class TodoModel {
  constructor () {
    this.STORE_KEY = 'todolist'
    this.todoData = localStorage.getItem(this.STORE_KEY)?JSON.parse(localStorage.getItem(this.STORE_KEY)):[]
    // 把监听函数放到这里，当数据模型发生变化的时候，可以执行这里的函数
    this.listeners = []

  }

  // 订阅函数
  subscribe=(listener)=>{
    this.listeners.push(listener)
  }
  // 触发器
  emit=()=>{
    this.listeners.forEach((listener) => listener())
  }
  addTodo = (title)=>{
    let todo = {
      id: new Date().getTime(),
      title: title,
      completed: false
    }
    let todoData = [...this.todoData, todo]
    this.saveData(this.STORE_KEY, todoData)
  }
  saveData = (key, data) => {
    localStorage.setItem(key, JSON.stringify(data))
    this.todoData = data
    this.emit()
  }
  removeTodo = (id) => {
    let todoData = this.todoData.filter((item, index) => {
      // 返回true的保留，返回false的删除
      return item.id !== id
    })
    this.saveData(this.STORE_KEY, todoData)
  }

  toggleTodo = (id) => {
    let todoData = this.todoData.map((item, index) => {
      if (item.id === id) {
        item.completed = !item.completed
      }
      return item
    })
    this.saveData(this.STORE_KEY, todoData)
  }
}

第二步，入口文件中引入TodoModel类，并且使用
